rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }

    function isStoryOwner() {
      return isSignedIn() && resource.data.userId == request.auth.uid;
    }

    function onlyStoryLikeMutation() {
      return request.resource.data.diff(resource.data).changedKeys().hasOnly(['likes']) &&
        request.resource.data.userId == resource.data.userId;
    }

    function onlyCommentReactionMutation() {
      return request.resource.data.diff(resource.data).changedKeys().hasOnly(['likes', 'reactions']) &&
        request.resource.data.userId == resource.data.userId;
    }

    match /users/{userId} {
      allow read: if isOwner(userId);
      allow create: if isOwner(userId);
      allow update: if isOwner(userId);
      allow delete: if isOwner(userId);

      match /diaryEntries/{entryId} {
        allow read, create, update, delete: if isOwner(userId);
      }

      match /stories/{storyId} {
        allow read, create, update, delete: if isOwner(userId);
      }

      match /private/{docId} {
        allow read, create, update, delete: if isOwner(userId);
      }

      match /cartItems/{itemId} {
        allow read, create, update, delete: if isOwner(userId);
      }
    }

    match /stories/{storyId} {
      allow read: if true;
      allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;
      allow update: if isStoryOwner() || onlyStoryLikeMutation();
      allow delete: if isStoryOwner();

      match /comments/{commentId} {
        allow read: if true;
        allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;
        allow update: if (isSignedIn() && resource.data.userId == request.auth.uid) || onlyCommentReactionMutation();
        allow delete: if isSignedIn() && resource.data.userId == request.auth.uid;
      }
    }

    match /agencies/{agencyId} {
      allow read: if true;
      allow create, update, delete: if false;
    }

    match /experiences/{experienceId} {
      allow read: if true;
      allow create, update, delete: if false;
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
